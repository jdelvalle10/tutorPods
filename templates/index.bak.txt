    <h2 style="text-align: center; color: #333;">Doral Academy AI Tutor</h2>

    <div id="input-container">
        <input type="text" id="user-input" placeholder="Ask a question..." onkeypress="handleKeyPress(event)">
        <button onclick="sendMessage()">Send</button>
    </div>
</div>

<script>
    // This array stores the conversational memory
    let conversationHistory = [];

    async function sendMessage() {
        const inputField = document.getElementById("user-input");
        const chatBox = document.getElementById("chat-box");
        const message = inputField.value.trim();
        if (!message) return;

        // Display user message in UI
        chatBox.innerHTML += `<div class="message user-msg">${message}</div>`;
        inputField.value = "";
        chatBox.scrollTop = chatBox.scrollHeight;

        // Add the new user message to the history array
        conversationHistory.push({ "role": "user", "content": message });

        try {
            // Send the entire conversation history to the Python server
            const response = await fetch("/chat", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ messages: conversationHistory })
            });

            const data = await response.json();

            if (response.ok) {
                // Display AI response and save it to the history array
                chatBox.innerHTML += `<div class="message ai-msg">${data.response}</div>`;
                conversationHistory.push({ "role": "assistant", "content": data.response });
            } else {
                // If blocked by guardrails, display error but remove the bad prompt from history
                chatBox.innerHTML += `<div class="message error-msg">Alert: ${data.response}</div>`;
                conversationHistory.pop(); 
            }
        } catch (error) {
            chatBox.innerHTML += `<div class="message error-msg">Error connecting to the AI Tutor.</div>`;
            conversationHistory.pop();
        }
        chatBox.scrollTop = chatBox.scrollHeight;
    }

    function handleKeyPress(event) {
        if (event.key === "Enter") sendMessage();
    }
</script>
